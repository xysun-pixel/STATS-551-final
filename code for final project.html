# 计算每个州的人均收入
ave_agi <- agidf %>%
  group_by(STATE) %>%
  summarise(
    total_population = sum(N2, na.rm = TRUE),  # 计算每个州的总人口
    total_income = sum(A00100, na.rm = TRUE),      # 计算每个州的总收入
    total_tax = sum(A18300, na.rm = TRUE) 
  ) %>%
  mutate(per_capita_tax = total_tax/total_income)
